62 void
63 second_grid(Triangulation<2> &triangulation)
64 {
65  const Point<2> center(1, 0);
66  const double inner_radius = 0.5, outer_radius = 1.0;
67  GridGenerator::hyper_shell(
68  triangulation, center, inner_radius, outer_radius, 10);
69 
70  // triangulation.reset_all_manifolds();
71 
72  for (unsigned int step = 0; step < 5; ++step)
73  {
74  {
75  std::ofstream out("grid-2-" + std::to_string(step) + ".vtk");
76  GridOut grid_out;
77  grid_out.write_vtk(triangulation, out);
78  }
79 
80  {
81  std::ofstream out("grid-2-" + std::to_string(step) + ".svg");
82  GridOut grid_out;
83  grid_out.write_svg(triangulation, out);
84  }
85 
86  for (auto &cell : triangulation.active_cell_iterators())
87  {
88  for (const auto v : cell->vertex_indices())
89  {
90  const double distance_from_center =
91  center.distance(cell->vertex(v));
92 
93  if (std::fabs(distance_from_center - inner_radius) <=
94  1e-6 * inner_radius)
95  {
96  cell->set_refine_flag();
97  break;
98  }
99  }
100  }
101 
102  triangulation.execute_coarsening_and_refinement();
103  }
104

122 // `third_grid.vtk`. Refine the L-shaped mesh adaptively around the re-entrant
123 // corner three times (after the global refinement you already did), but with a
124 // twist: refine all cells with the distance between the center of the cell and
125 // re-entrant corner is smaller than 1/3.
126 void
127 third_grid(Triangulation<2> &triangulation)
128 {
129  GridGenerator::hyper_L(triangulation);
130 
131  std::cout << "Number of original vertices: " << triangulation.n_vertices()
132  << std::endl;
133 
134  triangulation.refine_global(1);
135 
136  const Point<2> center(0, 0);
137  for (unsigned int step = 0; step < 3; ++step)
138  {
139  {
140  std::ofstream out("grid-3-" + std::to_string(step) + ".vtk");
141  GridOut grid_out;
142  grid_out.write_vtk(triangulation, out);
143  }
144  {
145  std::ofstream out("grid-3-" + std::to_string(step) + ".svg");
146  GridOut grid_out;
147  grid_out.write_svg(triangulation, out);
148  }
149 
150  for (auto &cell : triangulation.active_cell_iterators())
151  {
152  const double distance_from_center = center.distance(cell->center());
153  if (std::fabs(distance_from_center) < 2.0 / 3.0)
154  cell->set_refine_flag();
155  }
156 
157  triangulation.execute_coarsening_and_refinement();
158  }
159  {
160  std::ofstream out("grid-3.svg");
161  GridOut grid_out;
162  grid_out.write_svg(triangulation, out);
163  }
164  {
165  std::ofstream out("grid-3.vtk");
166  GridOut grid_out;
167  grid_out.write_vtk(triangulation, out);
168  }
169 }
